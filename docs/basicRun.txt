4. Kick off
Step 1: Issue Upload command
 $r = curl.exe -s -X POST http://localhost:8000/api/uploads -F "file=@test.csv" | ConvertFrom-Json                  
Step 2: Get ID for refernce
$id = $r.upload_id 
Step 3: Get summary of upload_id
curl.exe -s "http://localhost:8000/api/uploads/$id/validation" | ConvertFrom-Json
Step 4: Get IDs for human review
curl.exe -s "http://localhost:8000/api/sar?upload_id=$id&status=pending_review" | ConvertFrom-Json | Select-Object -ExpandProperty items | Format-Table id, transaction_id, status, content
Step 5: Approve/Reject for each id
curl.exe -s -X PATCH "http://localhost:8000/api/sar/df8af7a9-0e74-440c-bf27-ee045806effa/review" -H "Content-Type: application/json" -d '{\"action\":\"approve\",\"notes\":\"Approved\"}' 
or 
To approve everything
$ids = curl.exe -s "http://localhost:8000/api/sar?upload_id=$id&status=pending_review" | ConvertFrom-Json | Select-Object -ExpandProperty items
$ids | ForEach-Object { 
  $body = @{action="approve"; notes="Approved"}
  $r = Invoke-RestMethod -Uri "http://localhost:8000/api/sar/$($_.id)/review" -Method PATCH -ContentType "application/json" -Body ($body | ConvertTo-Json -Compress)
  Write-Host "$($_.id): $($r.status)"
}
